Vertical sliding carousel

  • 14+
  • 5+
  • 4+
  • 10+

HTML

<div class="carousel vertical">
    <div class="items">
        <img src="one.jpg" />
        <img src="two.jpg" />
        <img src="three.jpg" />
        <img src="four.jpg" />
     </div>
</div>

CSS

.vertical {
    height: 500px; // height of one carousel item
    width: 333px; // width of one carousel item
    position: relative;
    overflow: hidden;
}
.vertical .items {
    -webkit-animation: vscroll 16s infinite;
    -moz-animation: vscroll 16s infinite;
    -ms-animation: vscroll 16s infinite;
}
// keyframes: 3s visibile + 1s transition for each item
@-webkit-keyframes vscroll {
    0%, 18.75%  { margin-top: 0 }
    25%, 43.75% { margin-top: -500px; }
    50%, 68.75% { margin-top: -1000px; }
    75%, 93.75% { margin-top: -1500px; }            
    100%  { margin-top: 0 }
}
@-moz-keyframes vscroll {
    0%, 18.75%  { margin-top: 0 }
    25%, 43.75% { margin-top: -500px; }
    50%, 68.75% { margin-top: -1000px; }
    75%, 93.75% { margin-top: -1500px; }            
    100%  { margin-top: 0 }
}
@-ms-keyframes vscroll {
    0%, 18.75%  { margin-top: 0 }
    25%, 43.75% { margin-top: -500px; }
    50%, 68.75% { margin-top: -1000px; }
    75%, 93.75% { margin-top: -1500px; }            
    100%  { margin-top: 0 }
}

Carousel with images dissolving through color

  • 14+
  • 5+
  • 4+
  • 10+

HTML

<div class="carousel colorDissolve">
    <img src="one.jpg" />
    <img src="two.jpg" />
    <img src="three.jpg"/>
    <img src="four.jpg" />
</div>

CSS

.colorDissolve {
    position:relative;
    overflow: hidden;
    width:333px; // width of one carousel item
    height: 500px; // height of one carousel item
    background: #000; // color to dissolve through
}
.colorDissolve img {
    position:absolute;
    left: 0;
    right:0;
    opacity: 0;
    -webkit-animation: dissolve 20s linear infinite;
    -moz-animation: dissolve 20s linear infinite;
    -ms-animation: dissolve 20s linear infinite;
}
.colorDissolve img:nth-child(2) {
    -webkit-animation-delay: 5s;
    -moz-animation-delay: 5s;
    -ms-animation-delay: 5s;
}
.colorDissolve img:nth-child(3) {
    -webkit-animation-delay: 10s;
    -moz-animation-delay: 10s;
    -ms-animation-delay: 10s;
}        
.colorDissolve img:nth-child(4) {
    -webkit-animation-delay: 15s;
    -moz-animation-delay: 15s;
    -ms-animation-delay: 15s;
}
// keyframes: 4.5s visibile + 0.5s transition for each item
@-webkit-keyframes dissolve {
    2.5%, 22.5% { opacity: 1;}    
    25%, 100% { opacity: 0; }
}
@-moz-keyframes dissolve {
    2.5%, 22.5% { opacity: 1;}    
    25%, 100% { opacity: 0; }
}    
@-ms-keyframes dissolve {
    2.5%, 22.5% { opacity: 1;}    
    25%, 100% { opacity: 0; }
}
