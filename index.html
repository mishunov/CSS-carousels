LESS

/* vertical */
@width:287px; // width of the carousel
@height: 430px; // height of the carousel
@numberOfItems: 4; // the number of items in the carousel
@visibleItemDuration: 4s; // time during which a carousel item is fully visible
@transitionDuration: 1s; // duration of the transition between two carousel items

@carouselPeriod: @numberOfItems * (@visibleItemDuration + @transitionDuration);

.vertical {
    width: @width;
    height: @height;
    position: relative;
    overflow: hidden;
    .items {
        -webkit-animation: vscroll @carouselPeriod infinite;
        -moz-animation: vscroll @carouselPeriod infinite;
        -ms-animation: vscroll @carouselPeriod infinite;
        animation: vscroll @carouselPeriod infinite;
    }
}

Carousel with images dissolving through color

  • 14+
  • 5+
  • 4+
  • 10+

Whaat? It doesn't work in IE6? Grab jQuery plugin for this carousel to extend the browser support to Opera, IE6+ and so on.

HTML

<div class="carousel colorDissolve">
    <img class="item" src="one.jpg" />
    <img class="item" src="two.jpg" />
    <img class="item" src="three.jpg"/>
    <img class="item" src="four.jpg" />
</div>

CSS

/* colorDissolve  */
.colorDissolve {
  position: relative;
  overflow: hidden;
  width: 287px;
  height: 430px;
  background: #000000;
}
.colorDissolve .item {
  position: absolute;
  left: 0;
  right: 0;
  opacity: 0;
  -webkit-animation: colorDissolve 24s linear infinite;
  -moz-animation: colorDissolve 24s linear infinite;
  -ms-animation: colorDissolve 24s linear infinite;
  animation: colorDissolve 24s linear infinite;
}
.colorDissolve .item:nth-child(2) {
  -webkit-animation-delay: 6s;
  -moz-animation-delay: 6s;
  -ms-animation-delay: 6s;
  animation-delay: 6s;
}
.colorDissolve .item:nth-child(3) {
  -webkit-animation-delay: 12s;
  -moz-animation-delay: 12s;
  -ms-animation-delay: 12s;
  animation-delay: 12s;
}
.colorDissolve .item:nth-child(4) {
  -webkit-animation-delay: 18s;
  -moz-animation-delay: 18s;
  -ms-animation-delay: 18s;
  animation-delay: 18s;
}

/* The keyframes calculations are based on assumption of 4 items in the carousel.
/* You should notice the pattern for your own calculations.

LESS

/* colorDissolve  */
@width:287px; // width of the carousel
@height: 430px; // height of the carousel
@numberOfItems: 4; // the number of items in the carousel
@visibleItemDuration: 4s; // time during which a carousel item is fully visible
@transitionDuration: 1s; // duration of the transition between two carousel items
@dissolveColor: #000; // a color to dissolve through

.colorDissolve {
    position:relative;
    overflow: hidden;
    width:@width;
    height: @height;
    background: @dissolveColor;

    .item {
        position:absolute;
        left: 0;
        right:0;
        opacity: 0;

        -webkit-animation: colorDissolve (@visibleItemDuration + 2 * @transitionDuration)*@numberOfItems linear infinite;
        -moz-animation: colorDissolve (@visibleItemDuration + 2 * @transitionDuration)*@numberOfItems linear infinite;
        -ms-animation: colorDissolve (@visibleItemDuration + 2 * @transitionDuration)*@numberOfItems linear infinite;
        animation: colorDissolve (@visibleItemDuration + 2 * @transitionDuration)*@numberOfItems linear infinite;

        &:nth-child(2) {
            -webkit-animation-delay: @visibleItemDuration + 2 * @transitionDuration;
            -moz-animation-delay: @visibleItemDuration + 2 * @transitionDuration;
            -ms-animation-delay: @visibleItemDuration + 2 * @transitionDuration;
            animation-delay: @visibleItemDuration + 2 * @transitionDuration;
        }
        &:nth-child(3) {
            -webkit-animation-delay: (@visibleItemDuration + 2 * @transitionDuration)*2;
            -moz-animation-delay: (@visibleItemDuration + 2 * @transitionDuration)*2;
            -ms-animation-delay: (@visibleItemDuration + 2 * @transitionDuration)*2;
            animation-delay: (@visibleItemDuration + 2 * @transitionDuration)*2;
        }
        &:nth-child(4) {
            -webkit-animation-delay: (@visibleItemDuration + 2 * @transitionDuration)*3;
            -moz-animation-delay: (@visibleItemDuration + 2 * @transitionDuration)*3;
            -ms-animation-delay: (@visibleItemDuration + 2 * @transitionDuration)*3;
            animation-delay: (@visibleItemDuration + 2 * @transitionDuration)*3;
        }
    }
}
